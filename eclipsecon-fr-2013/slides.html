Consise OO

case class Time(hours: Int = 0, minutes: Int = 0) {
  val asMinutes: Int =
    hours * 60 + minutes

  def minus(that: Time): Int =
    this.asMinutes - that.asMinutes

  def -(that: Time): Int =
    minus(that)

  override lazy val toString: String =
    f"$hours%02d:$minutes%02d"
}

Powerfull collections

  val times = List(Time(14, 30), Time(9, 0), Time(11, 30), Time(4, 0))
                            //> times  : List[Time] = List(10:30, 09:00, 11:30, 04:00)

  times.size                //> res6: Int = 4
  times.isEmpty             //> res7: Boolean = false

  times filter { t =>
    t.hours >= 10
  }                         //> res9: List[Time] = List(10:30, 11:30)

  times map { _.hours }     //> res10: List[Int] = List(10, 9, 11, 4)
  
  times groupBy {_.minutes}                       
                            //> res11: Map[Int,List[Time]] = 
                            //| Map(30 -> List(10:30, 11:30), 0 -> List(09:00, 04:00))

Future and Promise

  • Future
    • object holding a value which may become available at some point
    • will complete with success or failure
  • Promise
    • returns a Future
    • completes the future
  • Non-blocking by default
